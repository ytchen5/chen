| 符号  | 含义                             | 举例                             |
| ----- | -------------------------------- | -------------------------------- |
| ()    | 子shell中执行                    | (ls)                             |
| (())  | 数值比较，运算，C语言风格的      | ((23>22))                        |
| $()   | 命令替换                         | echo $(date +%F)                 |
| $(()) | 整数运算                         | echo $((1+3))                    |
| {}    | 集合                             | touch {1..3}.file                |
| ${}   | 变量引用（引用变量扩展功能常用） | echo ${url#*e}，echo ${url:+aaa} |
| []    | 条件测试                         | [ $? -eq 0 ]                     |
| [[]]  | 条件测试，支持正则表达式         | [[ $num =~ ^[0-9]+$ ]]           |
| $[]   | 整数运算                         | $[1+3]                           |
